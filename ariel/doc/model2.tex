\section{Inferring the Absolute Magnitude of an Individual Galaxy from its Rotation Velocity}
\label{sec:infer_single_galaxy_y}

Assume we have already fit the TF model to a sample and obtained the posterior
\begin{equation}
p(\theta \mid \{\hat x_i,\hat y_i\}_{i=1}^N),
\qquad
\theta \equiv (\alpha,\beta,\sigma_{\text{int},x},\sigma_{\text{int},y}, \mu_{\mathrm{TF}}, \tau).
\end{equation}
We now consider a \emph{new} galaxy, denoted by ``$\star$'', for which we measure a rotation-velocity proxy
$\hat x_\star$ with uncertainty $\sigma_{x,\star}$, and we wish to infer its true absolute magnitude $y_\star$
(or predict what absolute magnitude we should expect given $\hat x_\star$ and the TF calibration).

\subsection{Posterior predictive distribution}
The desired quantity is the posterior predictive distribution for a new galaxy with
measured velocity proxy $\hat x_\star$ and uncertainty $\sigma_{x,\star}$:
\begin{equation}
p\!\left(y_\star \mid \hat x_\star, \sigma_{x,\star}, \{\hat x_i,\hat y_i\}\right)
=
\int d\theta\;
p\!\left(y_\star \mid \hat x_\star, \sigma_{x,\star}, \theta\right)\,
p\!\left(\theta \mid \{\hat x_i,\hat y_i\}\right).
\label{eq:posterior_predictive_y}
\end{equation}
Given posterior draws $\{\theta^{(m)}\}_{m=1}^M$, approximate this integral by
\begin{equation}
p\!\left(y_\star \mid \hat x_\star, \sigma_{x,\star}, \cdots\right)
\;\approx\;
\frac{1}{M}\sum_{m=1}^M
p\!\left(y_\star \mid \hat x_\star, \sigma_{x,\star}, \theta^{(m)}\right).
\end{equation}

\paragraph{Single-draw predictive $p(y_\star \mid \hat x_\star,\sigma_{x,\star},\theta)$.}
Introduce the latent Tully--Fisher magnitude $y_{\mathrm{TF},\star}$. Then
\begin{equation}
p\!\left(y_\star \mid \hat x_\star,\sigma_{x,\star},\theta\right)
=
\int dy_{\mathrm{TF},\star}\;
p\!\left(y_\star \mid y_{\mathrm{TF},\star},\theta\right)\,
p\!\left(y_{\mathrm{TF},\star} \mid \hat x_\star,\sigma_{x,\star},\theta\right).
\label{eq:pred_y_integral_yTF}
\end{equation}
Assume intrinsic scatter in $y$ at fixed $y_{\mathrm{TF}}$:
\begin{equation}
p\!\left(y_\star \mid y_{\mathrm{TF},\star},\theta\right)
=
\mathcal N\!\left(y_\star \mid y_{\mathrm{TF},\star},\,\sigma_{\mathrm{int},y}^2\right).
\label{eq:pred_y_given_yTF}
\end{equation}

To obtain $p(y_{\mathrm{TF},\star} \mid \hat x_\star,\sigma_{x,\star},\theta)$, marginalize the latent
true velocity proxy $x_\star$:
\begin{equation}
x_\star \mid y_{\mathrm{TF},\star},\theta
\sim
\mathcal N\!\left(\frac{y_{\mathrm{TF},\star}-\beta}{\alpha},\,\sigma_{\mathrm{int},x}^2\right),
\qquad
\hat x_\star \mid x_\star \sim \mathcal N\!\left(x_\star,\,\sigma_{x,\star}^2\right),
\end{equation}
which implies
\begin{equation}
\hat x_\star \mid y_{\mathrm{TF},\star},\theta
\sim
\mathcal N\!\left(\frac{y_{\mathrm{TF},\star}-\beta}{\alpha},\,\sigma_{x,\mathrm{tot}}^2\right),
\qquad
\sigma_{x,\mathrm{tot}}^2 \equiv \sigma_{\mathrm{int},x}^2+\sigma_{x,\star}^2 .
\label{eq:xhat_given_yTF}
\end{equation}

\paragraph{Transformed Normal (likelihood as a Gaussian in $y_{\mathrm{TF},\star}$).}
Define
\begin{equation}
\mu_L \equiv \beta+\alpha\hat x_\star,
\qquad
\sigma_L^2 \equiv \alpha^2\sigma_{x,\mathrm{tot}}^2,
\qquad
\sigma_L = |\alpha|\,\sigma_{x,\mathrm{tot}}.
\label{eq:muL_sigmaL_defs}
\end{equation}
Then the Normal likelihood in $\hat x_\star$ can be rewritten exactly as
\begin{equation}
\mathcal N\!\left(\hat x_\star \,\middle|\, \frac{y_{\mathrm{TF},\star}-\beta}{\alpha},\,\sigma_{x,\mathrm{tot}}^2\right)
=
|\alpha|\;
\mathcal N\!\left(y_{\mathrm{TF},\star}\mid \mu_L,\sigma_L^2\right).
\label{eq:likelihood_transformed_normal}
\end{equation}

Bayes' rule gives
\begin{align}
p\!\left(y_{\mathrm{TF},\star} \mid \hat x_\star,\sigma_{x,\star},\theta\right)
&=
\frac{
p\!\left(\hat x_\star \mid y_{\mathrm{TF},\star},\sigma_{x,\star},\theta\right)\;
p\!\left(y_{\mathrm{TF},\star}\mid\theta\right)
}{
p\!\left(\hat x_\star \mid \sigma_{x,\star},\theta\right)
}
\nonumber\\
&=
\frac{
\bigl[\,|\alpha|\,\mathcal N(y_{\mathrm{TF},\star}\mid \mu_L,\sigma_L^2)\bigr]\;
p(y_{\mathrm{TF},\star}\mid\theta)
}{
|\alpha|\int dy\;\mathcal N(y\mid \mu_L,\sigma_L^2)\;p(y\mid\theta)
}
\nonumber\\
&=
\frac{
\mathcal N(y_{\mathrm{TF},\star}\mid \mu_L,\sigma_L^2)\;
p(y_{\mathrm{TF},\star}\mid\theta)
}{
\int dy\;\mathcal N(y\mid \mu_L,\sigma_L^2)\;p(y\mid\theta)
},
\label{eq:bayes_yTF_transformed}
\end{align}
where the factor $|\alpha|$ cancels and $p(y\mid\theta)$ should be read as the prior for $y_{\mathrm{TF}}$.

\paragraph{Case A: Gaussian model for $y_{\mathrm{TF}}$.}
Assume
\begin{equation}
y_{\mathrm{TF}} \mid \theta \sim \mathcal N(\mu_{\mathrm{TF}},\tau^2).
\end{equation}

\subparagraph{Denominator.}
Using Eq.~\eqref{eq:bayes_yTF_transformed}, the denominator is the Gaussian--Gaussian convolution:
\begin{equation}
\int dy\;\mathcal N(y\mid \mu_L,\sigma_L^2)\;\mathcal N(y\mid \mu_{\mathrm{TF}},\tau^2)
=
\mathcal N\!\left(\mu_L \mid \mu_{\mathrm{TF}},\ \sigma_L^2+\tau^2\right).
\label{eq:denom_gauss_yspace}
\end{equation}
Equivalently, in $\hat x_\star$-space,
\begin{equation}
p\!\left(\hat x_\star\mid \sigma_{x,\star},\theta\right)
=
\mathcal N\!\left(
\hat x_\star \,\middle|\,
\frac{\mu_{\mathrm{TF}}-\beta}{\alpha},\;
\sigma_{x,\mathrm{tot}}^2+\frac{\tau^2}{\alpha^2}
\right)
=
|\alpha|\;\mathcal N\!\left(\mu_L \mid \mu_{\mathrm{TF}},\ \sigma_L^2+\tau^2\right).
\end{equation}

\subparagraph{Posterior for $y_{\mathrm{TF},\star}$.}
From Eq.~\eqref{eq:bayes_yTF_transformed},
\begin{equation}
p\!\left(y_{\mathrm{TF},\star}\mid \hat x_\star,\sigma_{x,\star},\theta\right)
\propto
\mathcal N(y_{\mathrm{TF},\star}\mid \mu_L,\sigma_L^2)\;
\mathcal N(y_{\mathrm{TF},\star}\mid \mu_{\mathrm{TF}},\tau^2),
\end{equation}
so
\begin{equation}
p\!\left(y_{\mathrm{TF},\star}\mid \hat x_\star,\sigma_{x,\star},\theta\right)
=
\mathcal N\!\left(y_{\mathrm{TF},\star}\mid \mu_\star,\ V_\star\right),
\label{eq:post_yTF_gauss}
\end{equation}
with
\begin{equation}
V_\star
=
\left(\frac{1}{\tau^2}+\frac{1}{\sigma_L^2}\right)^{-1},
\qquad
\mu_\star
=
V_\star\left(\frac{\mu_{\mathrm{TF}}}{\tau^2}+\frac{\mu_L}{\sigma_L^2}\right).
\label{eq:post_yTF_gauss_params}
\end{equation}

\subparagraph{Predictive for $y_\star$.}
Using Eq.~\eqref{eq:pred_y_given_yTF},
\begin{equation}
p\!\left(y_\star \mid \hat x_\star,\sigma_{x,\star},\theta\right)
=
\mathcal N\!\left(y_\star \mid \mu_\star,\ \sigma_{\mathrm{int},y}^2+V_\star\right).
\label{eq:pred_y_gauss}
\end{equation}

\paragraph{Case B: top-hat (uniform) model for $y_{\mathrm{TF}}$.}
Assume a uniform prior on an interval:
\begin{equation}
p(y_{\mathrm{TF}}\mid\theta)
=
\mathrm{Unif}(y_{\min},y_{\max})
=
\frac{1}{y_{\max}-y_{\min}}\;\mathbf 1_{[y_{\min},y_{\max}]}(y_{\mathrm{TF}}).
\end{equation}
Let $\Phi(\cdot)$ and $\varphi(\cdot)$ denote the standard normal CDF and PDF.

\subparagraph{Denominator.}
Using the transformed likelihood in Eq.~\eqref{eq:likelihood_transformed_normal},
\begin{align}
p\!\left(\hat x_\star\mid \sigma_{x,\star},\theta\right)
&=
\int dy\;
\mathcal N\!\left(\hat x_\star \,\middle|\, \frac{y-\beta}{\alpha},\ \sigma_{x,\mathrm{tot}}^2\right)\;
\frac{1}{y_{\max}-y_{\min}}\mathbf 1_{[y_{\min},y_{\max}]}(y)
\nonumber\\
&=
\frac{|\alpha|}{y_{\max}-y_{\min}}
\int_{y_{\min}}^{y_{\max}}
\mathcal N\!\left(y \mid \mu_L,\sigma_L^2\right)\,dy
\nonumber\\
&=
\frac{|\alpha|}{y_{\max}-y_{\min}}
\left[
\Phi\!\left(\frac{y_{\max}-\mu_L}{\sigma_L}\right)
-
\Phi\!\left(\frac{y_{\min}-\mu_L}{\sigma_L}\right)
\right].
\label{eq:denom_tophat}
\end{align}

\subparagraph{Posterior for $y_{\mathrm{TF},\star}$.}
From Eq.~\eqref{eq:bayes_yTF_transformed}, the posterior is proportional to
$\mathcal N(y_{\mathrm{TF},\star}\mid \mu_L,\sigma_L^2)$ restricted to $[y_{\min},y_{\max}]$; equivalently it
is a truncated normal:
\begin{equation}
p\!\left(y_{\mathrm{TF},\star}\mid \hat x_\star,\sigma_{x,\star},\theta\right)
=
\frac{
\frac{1}{\sigma_L}\,
\varphi\!\left(\frac{y_{\mathrm{TF},\star}-\mu_L}{\sigma_L}\right)\;
\mathbf 1_{[y_{\min},y_{\max}]}(y_{\mathrm{TF},\star})
}{
Z_L
},
\qquad
Z_L \equiv
\Phi\!\left(\frac{y_{\max}-\mu_L}{\sigma_L}\right)
-
\Phi\!\left(\frac{y_{\min}-\mu_L}{\sigma_L}\right),
\label{eq:post_yTF_tophat}
\end{equation}
where $\mu_L=\beta+\alpha\hat x_\star$ and $\sigma_L=|\alpha|\,\sigma_{x,\mathrm{tot}}$.

\subparagraph{Predictive for $y_\star$ (closed form).}
Let $\sigma_y^2 \equiv \sigma_{\mathrm{int},y}^2$ and define
\begin{equation}
\sigma_Z^2 \equiv \sigma_L^2+\sigma_y^2,
\qquad
\tilde\sigma^2 \equiv \left(\frac{1}{\sigma_L^2}+\frac{1}{\sigma_y^2}\right)^{-1},
\qquad
\tilde\mu(y_\star) \equiv \tilde\sigma^2\left(\frac{\mu_L}{\sigma_L^2}+\frac{y_\star}{\sigma_y^2}\right).
\label{eq:tophat_aux}
\end{equation}
Then Eq.~\eqref{eq:pred_y_integral_yTF} evaluates to
\begin{equation}
p\!\left(y_\star \mid \hat x_\star,\sigma_{x,\star},\theta\right)
=
\mathcal N\!\left(y_\star \mid \mu_L,\ \sigma_Z^2\right)\;
\frac{
\Phi\!\left(\frac{y_{\max}-\tilde\mu(y_\star)}{\tilde\sigma}\right)
-
\Phi\!\left(\frac{y_{\min}-\tilde\mu(y_\star)}{\tilde\sigma}\right)
}{
Z_L
},
\label{eq:pred_y_tophat}
\end{equation}
with $Z_L$ from Eq.~\eqref{eq:post_yTF_tophat}.
